[assembly: System.Reflection.AssemblyMetadata("RepositoryUrl", "https://github.com/Testably/Awaiten.git")]
[assembly: System.Runtime.Versioning.TargetFramework(".NETCoreApp,Version=v8.0", FrameworkDisplayName=".NET 8.0")]
namespace Awaiten
{
    public static class AwaitenResolverExtensions
    {
        extension(Awaiten.IAwaitenResolver resolver)
        {
            public T Resolve<T>() { }
            public bool TryResolve<T>([System.Diagnostics.CodeAnalysis.NotNullWhen(true)] out T? instance) { }
        }
    }
    [System.AttributeUsage(System.AttributeTargets.Class, Inherited=false)]
    public sealed class ContainerAttribute : System.Attribute
    {
        public ContainerAttribute() { }
        public bool SyncResolveAfterInit { get; set; }
    }
    public interface IAwaitenResolver
    {
        object Resolve(System.Type serviceType);
        bool TryResolve(System.Type serviceType, [System.Diagnostics.CodeAnalysis.NotNullWhen(true)] out object? instance);
    }
    public interface IAwaitenResolver<T>
    {
        T Resolve();
    }
    public interface IAwaitenScope : Awaiten.IAwaitenResolver, System.IDisposable
    {
        Awaiten.IAwaitenScope CreateScope();
    }
    [System.AttributeUsage(System.AttributeTargets.Class, AllowMultiple=true, Inherited=false)]
    public sealed class ScopedAttribute<TImplementation> : System.Attribute
        where TImplementation :  class
    {
        public ScopedAttribute() { }
        public string? Factory { get; set; }
    }
    [System.AttributeUsage(System.AttributeTargets.Class, AllowMultiple=true, Inherited=false)]
    public sealed class ScopedAttribute<TImplementation, TService> : System.Attribute
        where TImplementation :  class, TService
    {
        public ScopedAttribute() { }
        public string? Factory { get; set; }
    }
    [System.AttributeUsage(System.AttributeTargets.Class, AllowMultiple=true, Inherited=false)]
    public sealed class SingletonAttribute<TImplementation> : System.Attribute
        where TImplementation :  class
    {
        public SingletonAttribute() { }
        public string? Factory { get; set; }
        public string? Instance { get; set; }
    }
    [System.AttributeUsage(System.AttributeTargets.Class, AllowMultiple=true, Inherited=false)]
    public sealed class SingletonAttribute<TImplementation, TService> : System.Attribute
        where TImplementation :  class, TService
    {
        public SingletonAttribute() { }
        public string? Factory { get; set; }
        public string? Instance { get; set; }
    }
    [System.AttributeUsage(System.AttributeTargets.Class, AllowMultiple=true, Inherited=false)]
    public sealed class TransientAttribute<TImplementation> : System.Attribute
        where TImplementation :  class
    {
        public TransientAttribute() { }
        public string? Factory { get; set; }
    }
    [System.AttributeUsage(System.AttributeTargets.Class, AllowMultiple=true, Inherited=false)]
    public sealed class TransientAttribute<TImplementation, TService> : System.Attribute
        where TImplementation :  class, TService
    {
        public TransientAttribute() { }
        public string? Factory { get; set; }
    }
}
